   this.tabs = this.tabs.concat(otherWindow.tabs);
   return this;
};

   this.tabs.push('new tab'); // let's open a new tab for now
   return this;
};

// When you close a tab
Window.prototype.tabClose = function (index) {
   var preventTabs = this.tabs;
   // console.log(`preventTabs: ${preventTabs}`);
   var tabsBeforeIndex = preventTabs.slice(0, index); // get the tabs before the tab // splice used before
   // console.log(`tabsBeforeIndex: ${tabsBeforeIndex}`);
   var preventTabs = this.tabs;
   // console.log(`preventTabs: ${preventTabs}`);
   var tabsAfterIndex = preventTabs.slice(index + 1); // get the tabs after the tab // splice used before
   // console.log(`tabsAfterIndex: ${tabsAfterIndex}`);
   this.tabs = tabsBeforeIndex.concat(tabsAfterIndex); // join them together
   return this;
